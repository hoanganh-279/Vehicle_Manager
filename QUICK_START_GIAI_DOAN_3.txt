1. 🔔 TOAST NOTIFICATIONS
   ────────────────────────────────────────────────────────────
   Cách dùng:
   
   toast.success('Thao tác thành công!');
   toast.error('Có lỗi xảy ra!');
   toast.warning('Cảnh báo!');
   toast.info('Thông tin');
   
   ────────────────────────────────────────────────────────────

2. ⏳ LOADING OVERLAY
   ────────────────────────────────────────────────────────────
   Cách dùng:
   
   loading.show('Đang xử lý...');
   // ... do something
   loading.hide();
   
   ────────────────────────────────────────────────────────────

3. ❓ CONFIRM DIALOG
   ────────────────────────────────────────────────────────────
   Cách dùng:
   
   confirmAction('Bạn có chắc chắn?', () => {
       // On confirm
       console.log('Confirmed');
   }, () => {
       // On cancel
       console.log('Cancelled');
   });
   
   ────────────────────────────────────────────────────────────

4. 🌐 FETCH WITH LOADING
   ────────────────────────────────────────────────────────────
   Cách dùng:
   
   const result = await fetchWithLoading(
       '/api/endpoint',
       { method: 'POST', body: JSON.stringify(data) },
       'Đang lưu...'
   );
   
   if (result.success) {
       console.log(result.data);
   }
   
   ────────────────────────────────────────────────────────────

5. 💾 AUTO-SAVE INDICATOR
   ────────────────────────────────────────────────────────────
   Cách dùng:
   
   autoSave.saving();  // Đang lưu...
   autoSave.saved();   // Đã lưu
   autoSave.error();   // Lỗi lưu
   
   ────────────────────────────────────────────────────────────

6. ✅ FORM VALIDATION
   ────────────────────────────────────────────────────────────
   Cách dùng:
   
   // Manual
   if (validateForm(formElement)) {
       // Submit
   }
   
   // Auto (thêm attribute)
   <form data-validate>
       <input required>
       <button type="submit">Submit</button>
   </form>
   
   ────────────────────────────────────────────────────────────

7. 🎯 UTILITIES
   ────────────────────────────────────────────────────────────
   
   // Debounce
   const searchDebounced = debounce((query) => {
       // Search logic
   }, 300);
   
   // Smooth Scroll
   smoothScrollTo(element, offset);
   
   // Table Enhancement
   <table data-enhance>
       <thead>
           <th data-sortable>Name</th>
       </thead>
   </table>
   
   ────────────────────────────────────────────────────────────

{% block scripts %}
<script>
    // Đã có sẵn toast, loading, confirmAction, etc.
    
    // Example: Delete button
    function deleteItem(id) {
        confirmAction('Bạn có chắc chắn muốn xóa?', async () => {
            const result = await fetchWithLoading(
                `/api/items/${id}`,
                { method: 'DELETE' },
                'Đang xóa...'
            );
            
            if (result.success) {
                toast.success('Đã xóa thành công!');
                // Reload table or remove row
            }
        });
    }
    
    // Example: Save form
    async function saveForm(formData) {
        autoSave.saving();
        
        const result = await fetchWithLoading(
            '/api/save',
            {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(formData)
            },
            'Đang lưu...'
        );
        
        if (result.success) {
            autoSave.saved();
            toast.success('Lưu thành công!');
        } else {
            autoSave.error();
            toast.error('Lỗi lưu dữ liệu');
        }
    }
</script>
{% endblock %}
