// Intercept fetch để sửa giá
const originalFetch = window.fetch;
window.fetch = function(...args) {
    if (args[0] === '/parking/exit') {
        const body = JSON.parse(args[1].body);
        console.log('📤 Original payload:', body);
        // Thử sửa giá (sẽ bị Backend ignore)
        body.parking_fee = 1000;
        console.log('🔴 Tampered payload:', body);
        args[1].body = JSON.stringify(body);
    }
    return originalFetch.apply(this, args);
};

5. Bấm "Xác nhận thanh toán"

✅ KẾT QUẢ MONG ĐỢI:
   - Backend KHÔNG tin giá 1000 từ Frontend
   - Backend tự tính lại giá đúng
   - Console log: "💰 Tính phí xe #X: Yh x Z = [giá đúng]"
   - Xe bị tính phí đúng (không phải 1000)

┌─────────────────────────────────────────────────────────────────────────┐
│ BƯỚC 7: KIỂM TRA DATABASE                                               │
└─────────────────────────────────────────────────────────────────────────┘

Mở SQL Server Management Studio, chạy:

-- Xem xe vừa thanh toán
SELECT TOP 10
    id,
    license_plate AS 'Biển số',
    actual_fee AS 'Phí',
    payment_method AS 'Phương thức',
    payment_status AS 'Trạng thái',
    exit_time AS 'Giờ ra'
FROM vehicles
WHERE status = 'exited'
  AND payment_method IS NOT NULL
ORDER BY exit_time DESC;

✅ KẾT QUẢ MONG ĐỢI:
   - Thấy xe vừa thanh toán
   - payment_method = 'cash' (hoặc phương thức đã chọn)
   - payment_status = 'paid'
   - actual_fee = [giá đúng]
